164 std::array<std::array<double, 3>, 3>& orientation

166 // Set position (first 3 elements)

333 void Validate() const {
334 // Check if the number of blades is valid
335 if (n_blades < 1) {
336 throw std::runtime_error("Invalid number of blades. Must be at least 1.");
337 }
338
339 // Validate hub and nacelle data - should have exactly one mesh point each
340 if (hub.NumberOfMeshPoints() != 1 || nacelle.NumberOfMeshPoints() != 1) {
341 throw std::runtime_error("Hub and nacelle should have exactly one mesh point each.");
342 }
343
344 // Check if the number of blade roots matches the number of blades
345 if (blade_roots.NumberOfMeshPoints() != NumberOfBlades()) {
346 throw std::runtime_error("Number of blade roots does not match number of blades.");
347 }
348
349 // Check if the blade nodes to blade number mapping is valid - should be the same size
350 if (blade_nodes_to_blade_num_mapping.size() != blade_nodes.NumberOfMeshPoints()) {
351 throw std::runtime_error("Blade node to blade number mapping size mismatch.");
352 }
353
354 // Check if the node indices by blade are valid - should be same size as number of blades
355 if (node_indices_by_blade.size() != NumberOfBlades()) {
356 throw std::runtime_error("Node indices by blade size mismatch.");
357 }
358
359 // Check if the total number of blade nodes is valid - should be the same as the number of
360 // aggregrated blade nodes
361 const auto total_nodes = std::transform_reduce(
362 std::cbegin(node_indices_by_blade), std::cend(node_indices_by_blade), 0UL, std::plus{},
363 [](const auto& blade) {
364 return blade.size();
365 }
366 );
367 if (total_nodes != blade_nodes.NumberOfMeshPoints()) {

389 void SetBladeNodeMotion(
390 size_t blade_number, size_t node_number, const std::array<double, 7>& position,
391 const std::array<double, 6>& velocity, const std::array<double, 6>& acceleration
392 ) {
393 // Check if the blade and node numbers are within the valid range
394 if (blade_number >= NumberOfBlades() ||
395 node_number >= node_indices_by_blade[blade_number].size()) {
396 throw std::out_of_range("Blade or node number out of range.");
397 }
398
399 // Rotation to convert blade orientation
400 const auto r_x2z = Eigen::Quaternion<double>(
401 Eigen::AngleAxis<double>(std::numbers::pi / 2., Eigen::Matrix<double, 3, 1>::Unit(1))
402 );
403
404 // Original orientation
405 const auto r = Eigen::Quaternion<double>(position[3], position[4], position[5], position[6]);
406
407 // Converted orientation
408 const auto r_adi = r * r_x2z;
409
410 // Position with new orientation
411 const std::array<double, 7> position_new{position[0], position[1], position[2], r_adi.w(),
412 r_adi.x(), r_adi.y(), r_adi.z()};
413
414 const size_t node_index = node_indices_by_blade[blade_number][node_number];
415 blade_nodes.SetValues(node_index, position_new, velocity, acceleration);

538 void InitializeBlades(const TurbineConfig& tc) {
539 size_t i_blade{0};
540 size_t i_node_total{0};
541
542 // Initialize blade roots and nodes
543 for (const auto& blade : tc.blade_initial_states) {
544 // Initialize blade root
545 this->SetBladeRootMotion(
546 i_blade, blade.root_initial_position, std::array<double, 6>{0., 0., 0., 0., 0., 0.},
547 std::array<double, 6>{0., 0., 0., 0., 0., 0.}
548 );
549
550 // Initialize blade nodes
551 size_t i_node{0};
552 for (const auto& position : blade.node_initial_positions) {
553 node_indices_by_blade[i_blade].emplace_back(i_node_total);
554 blade_nodes_to_blade_num_mapping.emplace_back(static_cast<int32_t>(i_blade + 1));
555
556 this->SetBladeNodeMotion(
557 i_blade, i_node, position,
558 std::array<double, 6>{0., 0., 0., 0., 0., 0.}, // velocity
559 std::array<double, 6>{0., 0., 0., 0., 0., 0.} // acceleration
560 );
561 ++i_node;
562 ++i_node_total;
563 }
564 ++i_blade;
565 }

754 void SetupRotors(std::span<const TurbineConfig> turbine_configs) {
755 auto ADI_C_SetupRotor = lib_.get_function<void(
756 int*, int*, const float*, float*, double*, float*, double*, int*, float*, double*, int*,
757 float*, double*, int*, int*, char*
758 )>("ADI_C_SetupRotor");
759
760 // Loop through turbine configurations
761 int32_t turbine_number{0};
762 for (const auto& tc : turbine_configs) {
763 // Turbine number is 1 indexed i.e. 1, 2, 3, ...
764 ++turbine_number;
765
766 // Convert bool -> int32_t to pass to the Fortran routine
767 int32_t is_horizontal_axis_int = tc.is_horizontal_axis ? 1 : 0;
768
769 // Validate the turbine config
770 tc.Validate();
771
772 // Create new turbine data to store the updates
773 // Note: TurbineData and MeshData are validated during construction, so no need to
774 // perform additional checks here
775 turbines.emplace_back(tc);
776 auto& td = turbines.back();
777
778 // Call setup rotor for each turbine
779 ADI_C_SetupRotor(
780 &turbine_number, // input: current turbine number
781 &is_horizontal_axis_int, // input: 1: HAWT, 0: VAWT or cross-flow
782 tc.reference_position.data(), // input: turbine reference position
783 td.hub.position[0].data(), // input: initial hub position
784 td.hub.orientation[0][0].data(), // input: initial hub orientation
785 td.nacelle.position[0].data(), // input: initial nacelle position
786 td.nacelle.orientation[0][0].data(), // input: initial nacelle orientation
787 &td.n_blades, // input: number of blades
788 td.blade_roots.position[0].data(), // input: initial blade root positions
